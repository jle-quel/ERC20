API data structure
J'ai utiliser les mêmes noms pour les parametres que dans les prototypes des fonctions de l'ERC20



--------------------------------------------------------------------------------------------------
function totalSupply() public constant returns (uint);
--------------------------------------------------------------------------------------------------


Entrée : 
body: JSON.stringify({
    function: 'totalSupply',
  })


Sortie : 
{"totalSupply" : "42"}


--------------------------------------------------------------------------------------------------
function balanceOf(address tokenOwner) public constant returns (uint balance);
--------------------------------------------------------------------------------------------------

Entrée : 
body: JSON.stringify({
    function: 'balanceOf',
    tokenOwner: 'User1',
  })


Sortie : 
{"balance" : "42"}


--------------------------------------------------------------------------------------------------
function allowance(address tokenOwner, address spender) public constant returns (uint remaining);
--------------------------------------------------------------------------------------------------

Entrée : 
body: JSON.stringify({
    function: 'allowance',
    tokenOwner: 'User1',
    spender: 'User2',
  })


Sortie : 
{"remaining" : "42"}



--------------------------------------------------------------------------------------------------
function transfer(address to, uint tokens) public returns (bool success);
--------------------------------------------------------------------------------------------------

Entrée : 
body: JSON.stringify({
    function: 'transfer',
    to: 'User2',
    tokens: '42',
    by: 'User1',
  })

// Le parametre 'by' identifie celui qui dépense


Sortie : 
{"success" : true}



--------------------------------------------------------------------------------------------------

// Le parametre 'by' identifie celui qui approuve
